//Reverse the string
//logic function
function reverseString(userString) {

let revString = [];

//How array works
//let name= Hello
//name[0] = "H"
//name[4] = "o"
//last pos in an array name.lenght -1

//reverse a string using a for loop
for (let index = userString.length - 1 ; index >= 0; index--) {
    revString += userString[index];

}

return revString;
}
